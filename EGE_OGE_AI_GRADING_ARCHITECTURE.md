```python
    question_text: str,
    student_answer: str,
    criteria: dict,
    examples: list = None
) -> str:
    """
    –§–æ—Ä–º–∏—Ä—É–µ—Ç –ø—Ä–æ–º–ø—Ç –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏ –ï–ì–≠/–û–ì–≠ –æ—Ç–≤–µ—Ç–∞
    
    Args:
        task_number: "–ó–∞–¥–∞–Ω–∏–µ 27" (–ï–ì–≠ –†—É—Å—Å–∫–∏–π)
        question_text: –¢–µ–∫—Å—Ç –∑–∞–¥–∞–Ω–∏—è
        student_answer: –û—Ç–≤–µ—Ç —É—á–µ–Ω–∏–∫–∞
        criteria: Dict —Å –∫—Ä–∏—Ç–µ—Ä–∏—è–º–∏ –∏–∑ –§–ò–ü–ò
        examples: Few-shot –ø—Ä–∏–º–µ—Ä—ã (–æ–ø—Ü–∏–æ–Ω–∞–ª—å–Ω–æ, –¥–ª—è –ø–æ–≤—ã—à–µ–Ω–∏—è —Ç–æ—á–Ω–æ—Å—Ç–∏)
    """
    parts = [
        f"=== –ó–ê–î–ê–ù–ò–ï ===",
        f"–ù–æ–º–µ—Ä: {task_number}",
        f"–¢–µ–∫—Å—Ç: {question_text}",
        f"",
        f"=== –ö–†–ò–¢–ï–†–ò–ò –û–¶–ï–ù–ò–í–ê–ù–ò–Ø (–§–ò–ü–ò) ===",
    ]
    
    # –î–æ–±–∞–≤–ª—è–µ–º –∫–∞–∂–¥—ã–π –∫—Ä–∏—Ç–µ—Ä–∏–π
    for criterion_id, criterion_data in criteria.items():
        parts.append(f"\n{criterion_id}. {criterion_data['name']} (–º–∞–∫—Å. {criterion_data['max_points']} –±–∞–ª–ª–∞)")
        parts.append(f"–û–ø–∏—Å–∞–Ω–∏–µ: {criterion_data['description']}")
        parts.append("–£—Ä–æ–≤–Ω–∏:")
        for level in criterion_data['levels']:
            parts.append(f"  {level['points']} –±–∞–ª–ª(–æ–≤): {level['requirements']}")
    
    # Few-shot –ø—Ä–∏–º–µ—Ä—ã (–µ—Å–ª–∏ –µ—Å—Ç—å)
    if examples:
        parts.append(f"\n=== –ü–†–ò–ú–ï–†–´ –û–¶–ï–ù–ò–í–ê–ù–ò–Ø (–¥–ª—è –∫–∞–ª–∏–±—Ä–æ–≤–∫–∏) ===")
        for i, ex in enumerate(examples[:2], 1):  # –ú–∞–∫—Å 2 –ø—Ä–∏–º–µ—Ä–∞, —á—Ç–æ–±—ã –Ω–µ —Ä–∞–∑–¥—É–≤–∞—Ç—å —Ç–æ–∫–µ–Ω—ã
            parts.append(f"\n–ü—Ä–∏–º–µ—Ä {i}:")
            parts.append(f"–û—Ç–≤–µ—Ç: {ex['student_answer'][:200]}...")
            parts.append(f"–û—Ü–µ–Ω–∫–∞: {ex['correct_grading']}")
            parts.append(f"–û–±—ä—è—Å–Ω–µ–Ω–∏–µ: {ex['explanation']}")
    
    # –û—Ç–≤–µ—Ç —É—á–µ–Ω–∏–∫–∞
    parts.append(f"\n=== –û–¢–í–ï–¢ –£–ß–ï–ù–ò–ö–ê (–ø—Ä–æ–≤–µ—Ä—è–µ–º—ã–π) ===")
    parts.append(student_answer if student_answer.strip() else "(–ø—É—Å—Ç–æ–π –æ—Ç–≤–µ—Ç)")
    
    parts.append(f"\n=== –ò–ù–°–¢–†–£–ö–¶–ò–Ø ===")
    parts.append("–û—Ü–µ–Ω–∏ –æ—Ç–≤–µ—Ç —Å—Ç—Ä–æ–≥–æ –ø–æ —É–∫–∞–∑–∞–Ω–Ω—ã–º –∫—Ä–∏—Ç–µ—Ä–∏—è–º. –ó–∞–ø–æ–ª–Ω–∏ JSON —Å–æ score –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∫—Ä–∏—Ç–µ—Ä–∏—è.")
    
    return "\n".join(parts)
```
